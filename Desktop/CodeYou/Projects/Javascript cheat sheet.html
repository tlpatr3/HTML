JavaScript Basics Cheat Sheet

Variables

let name = "Alice";
const pi = 3.14;
var age = 30;

Data Types

  • string: "Hello"
  • number: 42, 3.14
  • boolean: true, false
  • null: intentionally empty
  • undefined: declared but not assigned
  • symbol: unique identifier
  • bigint: large integers

Conditionals

if (x > 10) {
    // do something
} else if (x === 10) {
    // something else
} else {
    // fallback
}

Loops

for (let i = 0; i < 5; i++) {
    console.log(i);
}

while (condition) {
    // loop
}

do {
    // run once
} while (condition);

Functions

function greet(name) {
    return "Hello " + name;
}

const add = (a, b) => a + b;

Arrays

let fruits = ["apple", "banana"];
fruits.push("orange");
fruits.pop();
fruits.shift();
fruits.unshift("kiwi");
fruits.length;
fruits[0];

Array Methods

arr.forEach(item => console.log(item));
arr.map(item => item * 2);
arr.filter(item => item > 5);
arr.find(item => item === 10);
arr.reduce((sum, num) => sum + num, 0);

Objects

let user = { name: "Bob", age: 30 };
user.name;
user.job = "Engineer";
delete user.age;

JSON

let json = JSON.stringify(user);
let obj = JSON.parse(json);

DOM & Events

document.getElementById("title").innerText = "Hello";
document.querySelector(".box").style.color = "blue";

document.getElementById("btn").addEventListener("click", function() {
    alert("Button clicked!");
});

Debugging

console.log("Debug info");
console.error("Error message");
debugger;

Ternary Operator

Type Operators
